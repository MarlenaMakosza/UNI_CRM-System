Table adres {
    id integer [pk]
    ulica varchar(200) [not null]
    numer_budynku varchar(20) [not null]
    numer_lokalu varchar(20)
    kod_pocztowy char(6) [not null]
    miejscowosc varchar(100) [not null]
    wojewodztwo varchar(50) [not null]
  }

  Table status_klienta {
    id integer [pk]
    kod varchar(50) [not null, unique]
    nazwa varchar(100) [not null]
  }

  Table klient {
    id integer [pk]
    nip varchar(10) [not null, unique]
    nazwa_firmy varchar(200) [not null]
    imie varchar(100) [not null]
    nazwisko varchar(100) [not null]
    stanowisko varchar(100) [not null]
    email varchar(255) [not null]
    telefon varchar(30) [not null]
    adres_id integer [not null]
    status_klienta_id integer [not null]
    created_at timestamp [not null]
  }

  Table region {
    id integer [pk]
    nazwa varchar(100) [not null, unique]
  }

  Table przedstawiciel_handlowy {
    id integer [pk]
    imie varchar(100) [not null]
    nazwisko varchar(100) [not null]
    email varchar(255) [not null, unique]
    telefon varchar(30) [not null]
    region_id integer [not null]
    data_zatrudnienia date [not null]
    aktywny boolean
    haslo_hash varchar(255) [not null]
    rola varchar(20) [not null, note: 'pracownik lub szef']
  }

  Table typ_zdarzenia {
    id integer [pk]
    nazwa varchar(100) [not null, unique]
  }

  Table typ_umowy {
    id integer [pk]
    nazwa varchar(100) [not null, unique]
  }

  Table produkt {
    id integer [pk]
    nazwa varchar(100) [not null]
    opis text
    cena decimal(12,2) [not null]
  }

  Table umowa {
    id integer [pk]
    klient_id integer [not null]
    typ_id integer [not null]
    przedstawiciel_id integer [not null]
    status varchar(50) [not null]
    data_od date [not null]
    data_do date
    created_at timestamp [not null]
    wartosc_umowy decimal(14,2) [not null]
  }

  Table zdarzenie {
    id integer [pk]
    klient_id integer [not null]
    przedstawiciel_id integer [not null]
    typ_id integer [not null]
    umowa_id integer
    data_planowana timestamp
    data_realizacji timestamp
    status varchar(50) [note: 'zaplanowane, zrealizowane, odwołane']
    opis text
    notatki text
    created_at timestamp [not null]
  }

  Table pozycja_umowy {
    id integer [pk]
    umowa_id integer [not null]
    produkt_id integer [not null]
    ilosc decimal(10,2) [not null]
    jednostka varchar(20) [not null]
    cena_jednostkowa decimal(12,2) [not null]
    wartosc decimal(14,2) [note: 'obliczane: ilosc * cena_jednostkowa']
  }

  // Relacje klient
  Ref: klient.adres_id > adres.id
  Ref: klient.status_klienta_id > status_klienta.id

  // Relacje przedstawiciel
  Ref: przedstawiciel_handlowy.region_id > region.id

  // Relacje zdarzenie
  Ref: zdarzenie.klient_id > klient.id
  Ref: zdarzenie.przedstawiciel_id > przedstawiciel_handlowy.id
  Ref: zdarzenie.typ_id > typ_zdarzenia.id
  Ref: zdarzenie.umowa_id > umowa.id

  // Relacje umowa
  Ref: umowa.klient_id > klient.id
  Ref: umowa.typ_id > typ_umowy.id
  Ref: umowa.przedstawiciel_id > przedstawiciel_handlowy.id

  // Relacje pozycja_umowy
  Ref: pozycja_umowy.umowa_id > umowa.id
  Ref: pozycja_umowy.produkt_id > produkt.id
